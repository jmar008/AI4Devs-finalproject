  # Crear .env.production con variables

SERVIDOR:
  ssh usuario@mcp.jorgemg.es
  cd /opt/dealaai && git clone <repo> && cd AI4Devs-finalproject
  git checkout v1.0.0-mvp
  # Copiar .env.production
  sudo certbot certonly --standalone -d mcp.jorgemg.es
  docker-compose -f docker-compose.production.yml up -d
  sleep 60
  docker-compose -f docker-compose.production.yml exec backend \
    python manage.py migrate --noinput
  docker-compose -f docker-compose.production.yml exec backend \
    python manage.py createsuperuser
  docker-compose -f docker-compose.production.yml exec backend \
    python manage.py collectstatic --noinput

VERIFICAR:
  https://mcp.jorgemg.es/login   ← Probar login

🆘 TROUBLESHOOTING RÁPIDO:
═══════════════════════════════════════════════════════════════════════════

    → docker-compose restart backend
    
  Frontend no carga?
    → docker-compose logs frontend | tail -50
    
  BD no conecta?
    → docker-compose exec db pg_isready
    
  Static files no cargan?
    → docker-compose exec backend python manage.py collectstatic --noinput
